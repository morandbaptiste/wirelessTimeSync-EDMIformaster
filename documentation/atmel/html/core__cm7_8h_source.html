01863 {

01871   return (
01872            ((PreemptPriority & (uint32_t)((1UL << (PreemptPriorityBits)) - 1UL)) << SubPriorityBits) |
01873            ((SubPriority     & (uint32_t)((1UL << (SubPriorityBits    )) - 1UL)))
