Detailed Description

A general purpose timer

Example: 

 // Count the time to toggle a LED

 #include "mbed.h"

 Timer timer;
 DigitalOut led(LED1);
 int begin, end;

 int main() {
     timer.start();
     begin = timer.read_us();
     led = !led;
     end = timer.read_us();
     printf("Toggle the led takes %d us", end - begin);
 }
