00956 static inline void rtc_count_enable_events(
00957     struct rtc_module *const module,
00958     struct rtc_count_events *const events)
00959 {
00960     /* Sanity check arguments */
00961     Assert(module);
00962     Assert(module->hw);
00963 
00964     Rtc *const rtc_module = module->hw;
00965 
00966     uint32_t event_mask = 0;
00967 
00968     /* Check if the user has requested an overflow event. */
00969     if (events->generate_event_on_overflow) {
00970         event_mask |= RTC_MODE0_EVCTRL_OVFEO;
00971     }
00972 
00973     /* Check if the user has requested any compare events. */
00974     for (uint8_t i = 0; i < RTC_NUM_OF_COMP16; i++) {
00975         if (events->generate_event_on_compare[i]) {
00976             event_mask |= RTC_MODE0_EVCTRL_CMPEO(1 << i);
00977         }
00978     }
00979 
00980     /* Check if the user has requested any periodic events. */
00981     for (uint8_t i = 0; i < 8; i++) {
00982         if (events->generate_event_on_periodic[i]) {
00983             event_mask |= RTC_MODE0_EVCTRL_PEREO(1 << i);
00984         }
00985     }
00986 
00987     /* Enable given event(s). */
00988     rtc_module->MODE0.EVCTRL.reg |= event_mask;
