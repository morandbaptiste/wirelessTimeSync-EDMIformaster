00194 {
00195   uint32_t result;
00196   int32_t s = 4 /*sizeof(v)*/ * 8 - 1; // extra shift needed at end
00197 
00198   result = value;                      // r will be reversed bits of v; first get LSB of v
00199   for (value >>= 1; value; value >>= 1)
00200   {
00201     result <<= 1;
00202     result |= value & 1;
00203     s--;
00204   }
00205   result <<= s;                       // shift when v's highest bits are zero
00206   return(result);
