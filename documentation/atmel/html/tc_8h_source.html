01116     /* Sanity check arguments */

01119     Assert(events);

01122 
01123     uint32_t event_mask = 0;

01125     if (events->invert_event_input == true) {
01126         event_mask |= TC_EVCTRL_TCINV;

01128 
01129     if (events->on_event_perform_action == true) {
01130         event_mask |= TC_EVCTRL_TCEI;
01131     }
01132 
01133     if (events->generate_event_on_overflow == true) {
01134         event_mask |= TC_EVCTRL_OVFEO;
01135     }
01136 
01137     for (uint8_t i = 0; i < NUMBER_OF_COMPARE_CAPTURE_CHANNELS; i++) {
01138         if (events->generate_event_on_compare_channel[i] == true) {
01139             event_mask |= (TC_EVCTRL_MCEO(1) << i);
01140         }
01141     }
01142 
01143     tc_module->COUNT8.EVCTRL.reg |= event_mask | events->event_action;
01144 }
