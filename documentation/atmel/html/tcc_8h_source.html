01903 static inline void tcc_stop_counter(
01904     const struct tcc_module *const module_inst)
01905 {
01906     /* Sanity check arguments */
01907     Assert(module_inst);
01908     Assert(module_inst->hw);

01911     Tcc *const tcc_module = module_inst->hw;
01912     uint32_t last_cmd;
01913 
01914     /* Wait until last command is done */
01915     do {
01916         while (tcc_module->SYNCBUSY.bit.CTRLB) {
01917             /* Wait for sync */
01918         }
01919         last_cmd = tcc_module->CTRLBSET.reg & TCC_CTRLBSET_CMD_Msk;
01920         if (last_cmd == TCC_CTRLBSET_CMD_NONE) {
01921             break;
01922         } else if (last_cmd == TCC_CTRLBSET_CMD_STOP) {
01923             /* Command have been issued */
01924             return;
01925         } else if (last_cmd == TCC_CTRLBSET_CMD_RETRIGGER) {
01926             /* Cancel RETRIGGER command and issue STOP */
01927             tcc_module->CTRLBCLR.reg = TCC_CTRLBCLR_CMD_Msk;
01928         }
01929     } while (1);
01930 
01931     /* Write command to execute */
01932     tcc_module->CTRLBSET.reg = TCC_CTRLBSET_CMD_STOP;
