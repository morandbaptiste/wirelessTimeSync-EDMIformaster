00150         {
00151         uint8_t val, cnt = 255;
00152 
00153             trx_reg_write(RG_TRX_STATE, CMD_FORCE_TRX_OFF);
00154             trx_reg_write(RG_IRQ_MASK, TRX_IRQ_PLL_LOCK);
00155             trx_reg_write(RG_TRX_STATE, CMD_PLL_ON);

00162                 {
00163                     break;

00165             }
